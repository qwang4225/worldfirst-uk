Get started in minutes

Everything you need to integrate WorldFirst's global payment infrastructure into your platform.

# Install the SDK
npm install @worldfirst/sdk

// Initialise the client
const { WorldFirst } = require('@worldfirst/sdk');

const wf = new WorldFirst({
clientId: process.env.WF_CLIENT_ID,
clientSecret: process.env.WF_SECRET,
environment: 'sandbox'
});

// Fetch live FX rate
const rate = await wf.fx.getRate({
from: 'GBP',
to: 'USD',
amount: 10000
});

console.log(rate.convertedAmount);
// 12,650.00 USD
